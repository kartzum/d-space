Trees
Tree. InOrder
 
	
	private static class InOrder {
        void run(TreeNode node) {
            if (node == null) {
                return;
            }
            run(node.left);
            System.out.print(node.val);
            run(node.right);
        }
    }
Tree. PreOrder
 
	
	private static class PreOrder {
        void run(TreeNode node) {
            if (node == null) {
                return;
            }
            System.out.print(node.val);
            run(node.left);
            run(node.right);
        }
    }
Tree. PostOrder
 
	
	private static class PostOrder {
        void run(TreeNode node) {
            if (node == null) {
                return;
            }
            run(node.left);
            run(node.right);
            System.out.print(node.val);
        }
    }
