static int getMaxLength(int arr[]) {
    int count = 0;
    int result = 0;
    for (int i = 0; i < arr.length; i++) {
        if (arr[i] == 0) {
            count = 0;
        } else {
            count += 1;
            result = Math.max(result, count);
        }
    }
    return result;
}
                        
                    
                        
static boolean areAnagram(char str1[], char str2[]) {
    if (str1.length != str2.length) {
        return false;
    }
    int m = 256;
    int[] count = new int[m];
    for (int i = 0; i < str1.length; i++) {
        count[str1[i]]++;
        count[str2[i]]--;
    }
    for (int i = 0; i < str1.length; i++) {
        if (count[i] != 0) {
            return false;
        }
    }
    return true;
}
