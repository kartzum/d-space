Tree Traversals (Inorder, Preorder and Postorder)
Binary Tree. InOrder
 
	
	class InOrder {
        void run(TreeNode node) {
            if (node == null) {
                return;
            }
            run(node.left);
            System.out.print(node.val);
            run(node.right);
        }
    }
Binary Tree. PreOrder
 
	
	class PreOrder {
        void run(TreeNode node) {
            if (node == null) {
                return;
            }
            System.out.print(node.val);
            run(node.left);
            run(node.right);
        }
    }
Binary Tree. PostOrder
 
	
	class PostOrder {
        void run(TreeNode node) {
            if (node == null) {
                return;
            }
            run(node.left);
            run(node.right);
            System.out.print(node.val);
        }
    }
